// ── CONFIGURACIÓN ───────────────────────────── const EMAIL_DESTINO = "mercadotecnia@tecnocontrol.com.mx, mercadotecniatecnocontrol@gmail.com"; const NOMBRE_HOJA = "Actividades"; const FILA_INICIO = 12; // ✅ YA INTEGRADO const LINK_SHEET = "https://docs.google.com/spreadsheets/d/18iNKYxSs_qK7IG3ppHiGjIsA57dbG7r0ioLlxB-6Ic4/edit"; // Índices const CI_ACTIVIDAD = 1; const CI_TIPO = 2; const CI_ENCARGADO = 6; const CI_ESTATUS = 8; const CI_DIAS_REST = 10; function enviarResumenSemanal() { const hoja = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(NOMBRE_HOJA); const datos = hoja.getDataRange().getValues(); const hoy = new Date(); const vencidas = []; const urgentes = []; const enProceso = []; const noIniciadas = []; const completadas = []; const ranking = {}; for (let i = FILA_INICIO - 1; i < datos.length; i++) { const fila = datos[i]; const actividad = String(fila[CI_ACTIVIDAD] || "").trim(); if (!actividad) continue; const estatus = String(fila[CI_ESTATUS] || "").toLowerCase(); const encargado = String(fila[CI_ENCARGADO] || "Sin asignar"); const tipo = String(fila[CI_TIPO] || ""); const diasRaw = fila[CI_DIAS_REST]; const diasRest = (typeof diasRaw === "number") ? diasRaw : 0; if (!ranking[encargado]) ranking[encargado] = { total:0, vencidas:0 }; ranking[encargado].total++; const item = { actividad, encargado, tipo, diasRest }; if (estatus.includes("finalizado") || estatus.includes("completad")) { completadas.push(item); continue; } if (diasRest < 0) { vencidas.push(item); ranking[encargado].vencidas++; } else if (diasRest <= 3) urgentes.push(item); else if (estatus.includes("no iniciado")) noIniciadas.push(item); else enProceso.push(item); } const total = vencidas.length + urgentes.length + enProceso.length + noIniciadas.length + completadas.length; const cumplimiento = total === 0 ? 0 : Math.round((completadas.length / total) * 100); let semaforo = "🟢 CONTROLADO"; if (vencidas.length > 5) semaforo = "🔴 CRÍTICO"; else if (vencidas.length > 0) semaforo = "🟡 EN RIESGO"; const top = [...vencidas, ...urgentes] .sort((a,b)=>a.diasRest-b.diasRest) .slice(0,5); const topHTML = top.map(t=>`
  • ${t.actividad} — ${t.encargado} (${t.diasRest} días)
    • `).join(""); const rankingHTML = Object.entries(ranking) .sort((a,b)=>b[1].vencidas - a[1].vencidas) .slice(0,5) .map(([n,d])=>` ${n} ${d.total} ${d.vencidas} `).join(""); const fechaStr = hoy.toLocaleDateString("es-MX", { weekday:"long", year:"numeric", month:"long", day:"numeric" }); const cuerpo = `

    📊 Dashboard Ejecutivo · Tecnocontrol

    ${fechaStr}

    ${semaforo}

    ${total}
    Total
    ${cumplimiento}%
    Cumplimiento
    ${vencidas.length}
    Vencidas
    ${urgentes.length}
    Riesgo
    Avance
    ${cumplimiento}%
    📂 Abrir tablero

    ⚠️ Prioridades

      ${topHTML}

    👤 Responsables

    ${rankingHTML}
    EncargadoTotalVencidas

    Reporte automático · Nivel CEO

    `; MailApp.sendEmail({ to: EMAIL_DESTINO, subject: `📊 REPORTE CEO — ${semaforo}`, htmlBody: cuerpo }); }
